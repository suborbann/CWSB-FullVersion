local function ApplyHighlight(plr)
    if plr == LocalPlayer then return end
    if not PlayerIsAlive(plr) then RemoveHighlight(plr); return end
    local char = plr.Character

    if Config.HighlightEnabled then
        if not HighlightObjects[plr] then
            local hl = Instance.new("Highlight")
            hl.FillColor        = Config.HighlightColor
            hl.OutlineColor     = Config.HighlightOutline
            hl.FillTransparency = Config.HighlightFillTrans
            hl.OutlineTransparency = 0
            hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
            hl.Adornee  = char
            hl.Parent   = CoreGui
            HighlightObjects[plr] = hl
        else
            local hl = HighlightObjects[plr]
            hl.FillColor        = Config.HighlightColor
            hl.OutlineColor     = Config.HighlightOutline
            hl.FillTransparency = Config.HighlightFillTrans
        end
    else
        if HighlightObjects[plr] then pcall(function() HighlightObjects[plr]:Destroy() end); HighlightObjects[plr]=nil end
    end

    if Config.ToolHighlightEnabled then
        local tool = GetEquippedTool(char)
        if tool then
            if not ToolHighlightObjects[plr] then
                local th = Instance.new("Highlight")
                th.FillColor        = Config.ToolHighlightColor
                th.OutlineColor     = Config.ToolHighlightColor
                th.FillTransparency = 0.3
                th.OutlineTransparency = 0
                th.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                th.Adornee  = tool
                th.Parent   = CoreGui
                ToolHighlightObjects[plr] = th
            else
                ToolHighlightObjects[plr].FillColor    = Config.ToolHighlightColor
                ToolHighlightObjects[plr].OutlineColor = Config.ToolHighlightColor
                ToolHighlightObjects[plr].Adornee      = tool
            end
        else
            if ToolHighlightObjects[plr] then pcall(function() ToolHighlightObjects[plr]:Destroy() end); ToolHighlightObjects[plr]=nil end
        end
    else
        if ToolHighlightObjects[plr] then pcall(function() ToolHighlightObjects[plr]:Destroy() end); ToolHighlightObjects[plr]=nil end
    end
end

local function RemoveESP(plr)
    if not ESP_Objects[plr] then return end
    for _, obj in pairs(ESP_Objects[plr]) do
        if typeof(obj)=="table" then
            for _,sub in ipairs(obj) do pcall(function() if sub and sub.Remove then sub:Remove() end end) end
        else pcall(function() if obj and obj.Remove then obj:Remove() end end) end
    end
    ESP_Objects[plr] = nil

-- ===================== WALL CHECK =====================
local function HasLOS(targetPos)
    if not Config.AimWallCheck then return true end
    local origin=Cam.CFrame.Position; local dir=(targetPos-origin); local dist=dir.Magnitude
    local ray=Ray.new(origin, dir.Unit*dist)
    local ignore={LocalPlayer.Character}
    for _,p in ipairs(Players:GetPlayers()) do if p.Character then table.insert(ignore,p.Character) end end
    local hit=Workspace:FindPartOnRayWithIgnoreList(ray,ignore)
    return hit==nil
end

local function ExpandPart(char)
    if not char then return end
    local hum=char:FindFirstChildOfClass("Humanoid")
    if hum and hum.Health<=0 then return end
    local part=char:FindFirstChild("UpperTorso"); if not part then return end
    if not OriginalSizes[part] then OriginalSizes[part]=part.Size end
    if not OriginalTransparencies[part] then OriginalTransparencies[part]=part.Transparency end
    part.Size=Vector3.new(Config.HitboxSize,Config.HitboxSize,Config.HitboxSize)
    part.Transparency=Config.HitboxTransparency
    -- FIX: no collision so it doesn't push players
    part.CanCollide=false
    part.Massless=true
end

local function RestorePart(char)

        part.Size=OriginalSizes[part]; OriginalSizes[part]=nil
        part.Transparency=OriginalTransparencies[part] or 0; OriginalTransparencies[part]=nil
        part.CanCollide=true
    end
end

local function SerializeConfig(t)
    local parts={}
    for k,v in pairs(t) do
        local sv=type(v)=="boolean" and (v and "1" or "0") or type(v)=="number" and tostring(v) or type(v)=="string" and v or "nil"
        table.insert(parts,k.."="..sv)
    end
    return table.concat(parts,";")
end
local function DeserializeConfig(s)
    local t={}
    for pair in s:gmatch("[^;]+") do
        local k,v=pair:match("^(.-)=(.+)$")
        if k and v then
            if v=="1" then t[k]=true elseif v=="0" then t[k]=false elseif v=="nil" then t[k]=nil elseif tonumber(v) then t[k]=tonumber(v) else t[k]=v end
        end
    end
    return t
end
local function LoadAllConfigs()
    local ok,raw=pcall(function() return readfile("CloudWare_configs.txt") end)
    if not ok or not raw or raw=="" then return {} end
    local configs={}
    for entry in raw:gmatch("[^\2]+") do
        local name,data=entry:match("^(.-)[\1](.+)$")
        if name and data then configs[name]=data end
    end
    return configs
end
local function SaveAllConfigs(configs)
    local parts={}
    for name,data in pairs(configs) do table.insert(parts,name.."\1"..data) end
    pcall(function() writefile("CloudWare_configs.txt",table.concat(parts,"\2")) end)
end

-- ===================== AIMBOT =====================
local function FindBestTarget(rx,ry)
    local bestDist,bestTarget,bestBone=math.huge,nil,nil
    for _,p in pairs(Players:GetPlayers()) do
        if p==LocalPlayer or IsWhitelisted(p) or not p.Character then continue end
        if not PlayerIsAlive(p) then continue end
        local char=p.Character; local bone=char:FindFirstChild("Head") or char:FindFirstChild("UpperTorso"); if not bone then continue end
        local pos,onScreen=Cam:WorldToViewportPoint(bone.Position)
        if not onScreen or pos.Z<0 then continue end
        local dist=(Vector2.new(rx,ry)-Vector2.new(pos.X,pos.Y)).Magnitude
        if dist<bestDist and dist<=Config.AimFOVSize and HasLOS(bone.Position) then
            bestDist=dist; bestTarget=p; bestBone=bone
        end
    end
    return bestTarget,bestBone
end
local function AimCamera(bone,smoothness)
    if not bone then return end
    Cam.CFrame=Cam.CFrame:Lerp(CFrame.new(Cam.CFrame.Position,bone.Position),math.clamp(smoothness/10,0.05,0.55))
end
local function GetAimLP(s) return 0.06+(1-(s-1)/4)*0.44 end
local function AimCursor(bone,cx,cy,smoothness)
    if not bone then return end
    local pos,onS=Cam:WorldToViewportPoint(bone.Position); if not onS or pos.Z<0 then return end
    local lp=GetAimLP(smoothness)
    if mousemoverel then mousemoverel((pos.X-cx)*lp,(pos.Y-cy)*lp)
    else Cam.CFrame=Cam.CFrame:Lerp(CFrame.new(Cam.CFrame.Position,bone.Position),lp) end
end

    if aimTrigger then
        if Config.AimMode=="Camera" then local _,bone=FindBestTarget(mouseX,mouseY); if bone then AimCamera(bone,Config.AimSmoothness) end
        else local _,bone=FindBestTarget(Mouse.X,mouseY); if bone then AimCursor(bone,Mouse.X,mouseY,Config.AimSmoothness) end end
    end

    -- Highlights at 10fps to save performance
    if now-lastHLUpdate>=0.1 then
        lastHLUpdate=now
        RefreshHighlights()
        UpdateSelfHighlight()
    end

    -- ESP: skip frame if interval not reached
    if now-lastESPUpdate<espInterval then return end
    lastESPUpdate=now

    -- Hide all first
    for _,esp in pairs(ESP_Objects) do pcall(function()
        if esp.Box then esp.Box.Visible=false end
        if esp.HBar then esp.HBar.Visible=false end
        if esp.NameTxt then esp.NameTxt.Visible=false end
        if esp.InvTxt then esp.InvTxt.Visible=false end
        if esp.TracerLine then esp.TracerLine.Visible=false end
        if esp.HeadCircle then esp.HeadCircle.Visible=false end
        if esp.SkeletonLines then for _,l in ipairs(esp.SkeletonLines) do l.Visible=false end end
    end) end

    -- Remove stale / dead players
    local toRm={}
    for plr in pairs(ESP_Objects) do
        local valid=false
        if plr and plr.Parent==Players then
            if PlayerIsAlive(plr) then valid=true end
        end
        if not valid then table.insert(toRm,plr) end
    end
    for _,plr in ipairs(toRm) do RemoveESP(plr) end

    if not Config.ESPVisible then espInterval=0.1; return end

    local visCount=0
    for _,p in pairs(Players:GetPlayers()) do
        if p==LocalPlayer then continue end
        if IsWhitelisted(p) then continue end
        -- Skip dead players
        if not PlayerIsAlive(p) then
            if ESP_Objects[p] then RemoveESP(p) end; continue
        end
        local char=p.Character
        local hum=char:FindFirstChildOfClass("Humanoid")
        local hrp=char:FindFirstChild("HumanoidRootPart"); local head=char:FindFirstChild("Head")
        if not hrp or not head or not char.Parent then continue end

        if Config.HitboxEnabled and not IsWhitelisted(p) then ExpandPart(char) end
        local rp,onS=Cam:WorldToViewportPoint(hrp.Position)
        if not onS or rp.Z<0 then continue end
        if (hrp.Position-Cam.CFrame.Position).Magnitude>1200 then continue end
        visCount+=1; if visCount>16 then continue end

        if not ESP_Objects[p] then
            ESP_Objects[p]={
                Box=Drawing.new("Square"),
                HBar=Drawing.new("Square"),
                NameTxt=Drawing.new("Text"),
                InvTxt=Drawing.new("Text"),
                TracerLine=Drawing.new("Line"),
                HeadCircle=Drawing.new("Circle"),
                SkeletonLines={}
            }
            local d=ESP_Objects[p]
            d.Box.Thickness=1; d.Box.Filled=false; d.Box.Color=Color3.new(1,1,1)
            d.HBar.Thickness=1; d.HBar.Filled=true
            for _,txt in ipairs({d.NameTxt,d.InvTxt}) do txt.Outline=false; txt.Color=Color3.new(1,1,1); txt.Font=2 end
            d.NameTxt.Center=true; d.NameTxt.Size=13; d.InvTxt.Center=true; d.InvTxt.Size=11
            d.TracerLine.Thickness=1; d.TracerLine.Color=Color3.new(1,1,1)
            d.HeadCircle.Thickness=1; d.HeadCircle.Filled=false; d.HeadCircle.Color=Color3.new(1,1,1)
        end

        local d=ESP_Objects[p]
        local topPos=Cam:WorldToViewportPoint(head.Position+Vector3.new(0,0.7,0))
        local botPos=Cam:WorldToViewportPoint(hrp.Position-Vector3.new(0,2.8,0))
        local h=math.abs(botPos.Y-topPos.Y); if h<1 then continue end
        local w=h*0.55; local bp=Vector2.new(rp.X-w/2,topPos.Y)

        if Config.BoxesEnabled then d.Box.Size=Vector2.new(w,h); d.Box.Position=bp; d.Box.Visible=true end
        if Config.HealthEnabled then
            local hr=math.clamp(hum.Health/hum.MaxHealth,0,1)
            d.HBar.Color=Color3.fromHSV(hr*0.33,1,1); d.HBar.Size=Vector2.new(3,h*hr)
            d.HBar.Position=Vector2.new(bp.X-6,bp.Y+h-(h*hr)); d.HBar.Visible=true
        end
        if Config.NamesEnabled then d.NameTxt.Text=p.Name; d.NameTxt.Position=Vector2.new(rp.X,bp.Y-16); d.NameTxt.Visible=true end
        if Config.InvEnabled then d.InvTxt.Text=getInventory(p); d.InvTxt.Position=Vector2.new(rp.X,bp.Y+h+4); d.InvTxt.Visible=true end
        if Config.TracersEnabled then
            d.TracerLine.From=Vector2.new(Cam.ViewportSize.X/2,Cam.ViewportSize.Y)
            d.TracerLine.To=Vector2.new(rp.X,rp.Y); d.TracerLine.Visible=true
        end
        if Config.HeadESPEnabled then
            local hp,hs=Cam:WorldToViewportPoint(head.Position)
            if hs and hp.Z>0 then d.HeadCircle.Position=Vector2.new(hp.X,hp.Y); d.HeadCircle.Radius=600/hp.Z; d.HeadCircle.Visible=true end
        end
        if Config.SkeletonEnabled and visCount<=8 then
            if #d.SkeletonLines==0 then
                for _=1,#bones do local ln=Drawing.new("Line"); ln.Color=Color3.new(1,1,1); ln.Thickness=1; ln.Transparency=1; table.insert(d.SkeletonLines,ln) end
            end
            for i,bone in ipairs(bones) do
                local b1=char:FindFirstChild(bone[1]); local b2=char:FindFirstChild(bone[2])
                if b1 and b2 then
                    local s1,v1=Cam:WorldToViewportPoint(b1.Position); local s2,v2=Cam:WorldToViewportPoint(b2.Position)
                    if v1 and v2 and s1.Z>0 and s2.Z>0 then
                        d.SkeletonLines[i].From=Vector2.new(s1.X,s1.Y); d.SkeletonLines[i].To=Vector2.new(s2.X,s2.Y); d.SkeletonLines[i].Visible=true
                    end
                end
            end
        end
    end
    -- Adaptive interval: fewer updates when many players visible
    espInterval = visCount>12 and 0.07 or visCount>6 and 0.05 or 0.04
end)

local function AddToggle(scroll, label, default, cb, bindKey, bindCb, colorKey, colorCb)
    rowOrd+=1
    local state=default or false
    local hasBind=bindKey~=nil
    local hasColor=colorKey~=nil
    local row=Instance.new("Frame",scroll); row.Size=UDim2.new(1,0,0,ROW_H); row.BackgroundColor3=C.ROW_BG; row.BorderSizePixel=0; row.LayoutOrder=rowOrd; row.ZIndex=7

    local function toggle()
        state=not state
        TweenService:Create(badge,TweenInfo.new(0.15),{BackgroundColor3=state and C.ON_BG or C.OFF_BG}):Play()
        bLbl.Text=state and "ON" or "OFF"; bLbl.TextColor3=state and C.ON_TEXT or C.OFF_TEXT
        if cb then cb(state) end
    end
    btn.MouseButton1Click:Connect(toggle)
    btn.InputBegan:Connect(function(i) if i.UserInputType==Enum.UserInputType.Touch then toggle() end end)
    if hasBind then
        local curBind=bindKey; local listeningBind=false
        local bb=Instance.new("Frame",row); bb.Size=UDim2.new(0,BINDW,0,BH); bb.Position=UDim2.new(1,bindX,0.5,-BH/2); bb.BackgroundColor3=C.BIND_BG; bb.BorderSizePixel=0; bb.ZIndex=11

                if i.UserInputType~=Enum.UserInputType.Keyboard then return end

                curBind=i.KeyCode.Name:sub(1,7); bbl.Text=curBind; bbl.TextColor3=C.BIND_TEXT; if bindCb then bindCb(i.KeyCode) end
            end)
            task.delay(8,function() if listeningBind then listeningBind=false; c:Disconnect(); bbl.Text=curBind end end)
        end
        bindBtn.MouseButton1Click:Connect(startBindListen)
        bindBtn.InputBegan:Connect(function(i) if i.UserInputType==Enum.UserInputType.Touch then startBindListen() end end)
    end
    if hasColor then
        local initCol=colorKey or Color3.new(1,0,0)
        local swatch=Instance.new("Frame",row); swatch.Size=UDim2.new(0,COLOR_SZ,0,COLOR_SZ); swatch.Position=UDim2.new(1,colorX,0.5,-COLOR_SZ/2)

                end)

        end

        end
    end
    return function(v)
        state=v; badge.BackgroundColor3=state and C.ON_BG or C.OFF_BG; bLbl.Text=state and "ON" or "OFF"; bLbl.TextColor3=state and C.ON_TEXT or C.OFF_TEXT
    end

-- Tab System
local tabs={}; local activeTab=nil; local NTABS=4; local TAB_W=math.floor(WIN_W/NTABS)
local function SelectTab(tabObj)
    if activeTab then activeTab.frame.Visible=false; activeTab.lbl.TextColor3=C.TAB_IDLE; activeTab.ind.Visible=false; activeTab.btn.BackgroundColor3=C.TABBAR_BG end
    activeTab=tabObj; tabObj.frame.Visible=true; tabObj.lbl.TextColor3=C.TAB_ACTIVE; tabObj.ind.Visible=true; tabObj.btn.BackgroundColor3=Color3.fromRGB(22,22,28)
end
local function CreateTab(name)
    local idx=#tabs
    local tabBtn=Instance.new("TextButton",TabBar); tabBtn.Size=UDim2.new(0,TAB_W,1,0); tabBtn.Position=UDim2.new(0,idx*TAB_W,0,0); tabBtn.BackgroundColor3=C.TABBAR_BG; tabBtn.BackgroundTransparency=0; tabBtn.Text=""; tabBtn.BorderSizePixel=0; tabBtn.ZIndex=9
    local ind=Instance.new("Frame",tabBtn); ind.Size=UDim2.new(0.75,0,0,2); ind.Position=UDim2.new(0.125,0,1,-2); ind.BackgroundColor3=C.WHITE; ind.BorderSizePixel=0; ind.Visible=false; ind.ZIndex=10; Instance.new("UICorner",ind).CornerRadius=UDim.new(1,0)
    local tabLbl=Instance.new("TextLabel",tabBtn); tabLbl.Size=UDim2.new(1,0,1,0); tabLbl.BackgroundTransparency=1; tabLbl.Text=name; tabLbl.TextColor3=C.TAB_IDLE; tabLbl.Font=Enum.Font.GothamBold; tabLbl.TextSize=IsSmall and 11 or 10; tabLbl.ZIndex=10
    local cf=Instance.new("Frame",ContentArea); cf.Size=UDim2.new(1,0,1,0); cf.BackgroundTransparency=1; cf.Visible=false; cf.ZIndex=6
    local sf=MakeScroll(cf)
    local tabObj={btn=tabBtn,frame=cf,scroll=sf,lbl=tabLbl,ind=ind}
    table.insert(tabs,tabObj)
    tabBtn.MouseButton1Click:Connect(function() SelectTab(tabObj) end)
    tabBtn.InputBegan:Connect(function(i) if i.UserInputType==Enum.UserInputType.Touch then SelectTab(tabObj) end end)
    tabBtn.MouseEnter:Connect(function() if activeTab~=tabObj then tabLbl.TextColor3=C.TAB_HOVER end end)
    tabBtn.MouseLeave:Connect(function() if activeTab~=tabObj then tabLbl.TextColor3=C.TAB_IDLE end end)
    return tabObj
end
